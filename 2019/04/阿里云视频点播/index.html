//当项目已启动，spring接口，spring加载之后，执行接口一个方法
@Component //把该类交给spring管理
public class ConstantUtils implements InitializingBean {

    @Value("${aliyun.vod.file.keyid}")
    private String keyId;
    @Value("${aliyun.vod.file.keysecret}")
    private String keySecret;

    //定义公开静态常量
    public static String ACCESS_KEY_ID;
    public static String ACCESS_KEY_SECRET;

    @Override
    public void afterPropertiesSet() throws Exception {
        ACCESS_KEY_ID = keyId;
        ACCESS_KEY_SECRET = keySecret;
    }
}

@RestController
@RequestMapping("/eduvod/video")
@CrossOrigin //解决跨域
public class VodController {

    @Autowired
    private VodService vodService;

    @ApiOperation("上传文件到阿里云")
    @PostMapping("uploadVideo")
    public R uploadFile(MultipartFile file) {
        //返回视频ID
        String videoId = vodService.uploadFile(file);
        return R.ok().data("videoId",videoId);
    }
    
    
    //该方法额外需要初始化类 
      @ApiOperation("删除阿里云的视频")
    @DeleteMapping("removeAlyVideo/{id}")
    public R removeAlyVideo(@PathVariable String id) {

        try {
            //初始化
            DefaultAcsClient client = InitVodClient.initVodClient(ConstantUtils.ACCESS_KEY_ID, ConstantUtils.ACCESS_KEY_SECRET);
            //创建删除视频的request对象
            DeleteVideoRequest request = new DeleteVideoRequest();
            //向 request设置视频 id,可以批量删除，id直接用 , 隔开即可
            request.setVideoIds(id);
            //调用初始化对象的方法实现删除
            client.getAcsResponse(request);
            return R.ok();
        } catch (Exception e) {
            e.printStackTrace();
            throw new GuliException(20001, "删除视频失败");
        }
    }
    
    
    //批量删除阿里云的视频
    @ApiOperation("批量删除阿里云视频")
    @DeleteMapping("removeVideoBatch")
    public R removeVideoBatch(@RequestParam List VideoListId) {
         try {
            //初始化
            DefaultAcsClient client = InitVodClient.initVodClient(ConstantUtils.ACCESS_KEY_ID, ConstantUtils.ACCESS_KEY_SECRET);
            //创建删除视频的request对象
            DeleteVideoRequest request = new DeleteVideoRequest();

            //使用的是 org.apache.commons.lang.StringUtils
            String videoIds = StringUtils.join(videoListId.toArray(), ",");

            //向 request设置视频 id,可以批量删除，id直接用 , 隔开即可
            request.setVideoIds(videoIds);
            //调用初始化对象的方法实现删除
            client.getAcsResponse(request);
             return R.ok();
        } catch (Exception e) {
            e.printStackTrace();
            throw new GuliException(20001, "删除视频失败");
        }
        
    }
}

service层

  public String uploadFile(MultipartFile file) {

        try {
            //fileName:上传文件的名称
            String fileName = file.getOriginalFilename();
            //title：上传之后显示的视频名称，去掉后缀名  xxx.mp4 => xxx
            String title = fileName.substring(0, fileName.lastIndexOf("."));
            //inputStream
            InputStream inputStream = file.getInputStream();
            UploadStreamRequest request = new UploadStreamRequest(ConstantUtils.ACCESS_KEY_ID, ConstantUtils.ACCESS_KEY_SECRET, title, fileName, inputStream);

            UploadVideoImpl uploader = new UploadVideoImpl();
            UploadStreamResponse response = uploader.uploadStream(request);

            String videoId = null;
            if (response.isSuccess()) {
                videoId = response.getVideoId() ;
            } else { //如果设置回调URL无效，不影响视频上传，可以返回VideoId同时会返回错误码。其他情况上传失败时，VideoId为空，此时需要根据返回错误码分析具体错误原因
                videoId = response.getVideoId();
            }
            return videoId;
        } catch (IOException e) {
            e.printStackTrace();
            throw new GuliException(20001, "上传失败");//自定义异常
        }
    }

初始化类 （删除方法所需要）


import com.aliyun.oss.ClientException;
import com.aliyuncs.DefaultAcsClient;
import com.aliyuncs.profile.DefaultProfile;

public class InitVodClient {

    public static DefaultAcsClient initVodClient(String accessKeyId, String accessKeySecret) throws ClientException {
        String regionId = "cn-shanghai";  // 点播服务接入区域,固定不能修改
        DefaultProfile profile = DefaultProfile.getProfile(regionId, accessKeyId, accessKeySecret);
        DefaultAcsClient client = new DefaultAcsClient(profile);
        return client;
    }
}
